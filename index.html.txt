// Scroll reveal with stagger
const revealSets = [
  { selector: '.exhibit-card', delay: 130 },
  { selector: '.gallery-item', delay: 100 },
  { selector: '.price-panel', delay: 90 },
  { selector: '.maker-frame, .transparency-card, .contact-card', delay: 0 },
];

const io = new IntersectionObserver((entries) => {
  entries.forEach(entry => {
    if (!entry.isIntersecting) return;
    const el = entry.target;
    const d = parseInt(el.dataset.revealDelay || 0);
    setTimeout(() => el.classList.add('visible'), d);
    io.unobserve(el);
  });
}, { threshold: 0.07 });

revealSets.forEach(({ selector, delay }) => {
  document.querySelectorAll(selector).forEach((el, i) => {
    el.dataset.revealDelay = i * delay;
    io.observe(el);
  });
});

// Pricing accordion
function togglePanel(headEl) {
  const panel = headEl.closest('.price-panel');
  const isOpen = panel.classList.contains('open');
  document.querySelectorAll('.price-panel').forEach(p => p.classList.remove('open'));
  if (!isOpen) panel.classList.add('open');
}

// Open featured on load
document.querySelector('.price-panel.featured')?.classList.add('open');

const pills = document.querySelectorAll('.nav-pill');

    }
  }, { threshold: 0.4 }).observe(s)
